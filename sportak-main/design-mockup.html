🗃 TypeScript Types & Prisma Schema

// prisma/schema.prisma
model User {
  id          String   @id @default(cuid())
  slackId     String   @unique
  name        String
  avatarUrl   String?
  role        Role     @default(PARTICIPANT)
  createdAt   DateTime @default(now())
  activities  ActivityLog[]
  createdLogs ActivityLog[] @relation("CreatedBy")
  auditEntries AuditEntry[]
}
model Challenge {
  id       String          @id @default(cuid())
  name     String
  type     ChallengeType
  status   ChallengeStatus @default(DRAFT)
  startsAt DateTime
  endsAt   DateTime
  rules    String
}
model ActivityCatalog {
  id           String        @id @default(cuid())
  name         String
  unit         String
  indexPerUnit Float
  maxPerDay    Float?
  challengeType ChallengeType
}
model ActivityLog {
  id          String   @id @default(cuid())
  userId      String
  challengeId String
  catalogId   String
  date        DateTime
  value       Float
  points      Float   ← stored on write
  createdById String
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt
}
model AuditEntry {
  id          String   @id @default(cuid())
  challengeId String
  actorId     String
  action      String
  payload     Json
  createdAt   DateTime @default(now())
}
enum Role            { PARTICIPANT ADMIN }
enum ChallengeType   { WINTER SUMMER }
enum ChallengeStatus { DRAFT ACTIVE CLOSED ARCHIVED }

📊 Scoring Engine (src/lib/scoring.ts)
▼ 
import { ActivityCatalog } from '@prisma/client'

export interface ScoreResult {
  points:      number          // final points stored in ActivityLog.points
  cappedAt:    number | null   // null if no cap applied
  explanation: string          // human-readable breakdown
}

/** Pure function — no DB access, safe to call in real-time preview */
export function calculatePoints(
  catalog: Pick<ActivityCatalog, 'indexPerUnit' | 'maxPerDay' | 'unit'>,
  value: number
): ScoreResult {
  const cappedValue = catalog.maxPerDay
    ? Math.min(value, catalog.maxPerDay)
    : value
  const points = cappedValue * catalog.indexPerUnit
  return {
    points,
    cappedAt: cappedValue < value ? cappedValue : null,
    explanation: `${cappedValue} ${catalog.unit} × ${catalog.indexPerUnit} b/${catalog.unit} = ${points} b`
  }
}
💡 Volána v POST /api/challenges/[id]/activities i v client komponentě ActivityForm pro live preview bez DB.
📋 Klíčová pravidla & Constraints
▼
🔐 Auth guard: middleware.ts chrání /admin/* — session.user.role musí být ADMIN, jinak redirect na /challenges
📜 Audit trail: každý PATCH a DELETE v admin endpointech musí vytvořit AuditEntry se snapshoty before/after v payload (JSON)
📊 Scoring on write: body se přepočítávají při každém zápisu (POST/PATCH), výsledek se ukládá do ActivityLog.points — žádný cron
🔒 Immutable history: výzvy ve stavu CLOSED nebo ARCHIVED jsou read-only — API vrátí 403 pro jakoukoliv write operaci
👤 Role v DB: PARTICIPANT | ADMIN je pole User.role v PostgreSQL — Slack nespravuje role, pouze identitu
🎨 Design tokens: používej CSS proměnné z :root bloku — primary: #006dff, dark: #272727, success: #00b87a, danger: #e03131
